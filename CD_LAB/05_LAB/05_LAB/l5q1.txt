---------------------------------------------
Question
---------------------------------------------
Write a FLEX program to count the number of vowels and consonants in the given input.


---------------------------------------------
Code (l5q1.l)
---------------------------------------------

%{
#include <stdio.h>

int vowelTotal = 0;
int consonantTotal = 0;
%}

%%
[aeiouAEIOU]                    { vowelTotal++; }
[b-df-hj-np-tv-zB-DF-HJ-NP-TV-Z] { consonantTotal++; }
.|\n                            { }
%%

int main()
{
    yylex();
    printf("Number of vowels: %d\n", vowelTotal);
    printf("Number of consonants: %d\n", consonantTotal);
    return 0;
}

int yywrap()
{
    return 1;
}


---------------------------------------------
How to Run (Linux)
---------------------------------------------

flex l5q1.l
gcc lex.yy.c -o l5q1 -lfl
./l5q1


---------------------------------------------
Sample Input
---------------------------------------------
Hello World


---------------------------------------------
Sample Output
---------------------------------------------
Number of vowels: 3
Number of consonants: 7
